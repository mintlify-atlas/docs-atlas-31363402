## Get Websites

Returns all websites that the authenticated user has access to, including personal websites and team websites.

### Endpoint

```
GET /api/websites
```

### Headers

<ParamField header="Authorization" type="string" required>
  Bearer token for authentication
</ParamField>

### Query Parameters

<ParamField query="page" type="number">
  Page number for pagination (default: 1)
</ParamField>

<ParamField query="pageSize" type="number">
  Number of results per page
</ParamField>

<ParamField query="search" type="string">
  Search term to filter websites by name or domain
</ParamField>

<ParamField query="includeTeams" type="string">
  Include websites from teams the user belongs to. Set to any value to enable.
</ParamField>

### Response

<ResponseField name="data" type="array">
  Array of website objects
  
  <Expandable title="website object">
    <ResponseField name="id" type="string">
      Website UUID
    </ResponseField>
    
    <ResponseField name="name" type="string">
      Website name (max 100 characters)
    </ResponseField>
    
    <ResponseField name="domain" type="string">
      Website domain (max 500 characters)
    </ResponseField>
    
    <ResponseField name="shareId" type="string">
      Optional public share ID for sharing analytics
    </ResponseField>
    
    <ResponseField name="userId" type="string">
      Owner user UUID (for personal websites)
    </ResponseField>
    
    <ResponseField name="teamId" type="string">
      Team UUID (for team websites)
    </ResponseField>
    
    <ResponseField name="createdAt" type="string">
      ISO 8601 timestamp of creation
    </ResponseField>
    
    <ResponseField name="createdBy" type="string">
      UUID of user who created the website
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="count" type="number">
  Total number of websites
</ResponseField>

<ResponseField name="page" type="number">
  Current page number
</ResponseField>

<ResponseField name="pageSize" type="number">
  Results per page
</ResponseField>

### Example Request

<CodeGroup>

```bash curl
curl https://your-umami-instance.com/api/websites?includeTeams=true \
  -H "Authorization: Bearer YOUR_TOKEN"
```

```javascript JavaScript
const response = await fetch(
  'https://your-umami-instance.com/api/websites?includeTeams=true',
  {
    headers: {
      'Authorization': 'Bearer YOUR_TOKEN',
    },
  }
);

const websites = await response.json();
```

```python Python
import requests

response = requests.get(
    'https://your-umami-instance.com/api/websites',
    params={'includeTeams': 'true'},
    headers={'Authorization': 'Bearer YOUR_TOKEN'}
)

websites = response.json()
```

</CodeGroup>
