## Get Links

Retrieve all short links that the authenticated user has created or has access to through teams.

### Endpoint

```
GET /api/links
```

### Headers

<ParamField header="Authorization" type="string" required>
  Bearer token for authentication
</ParamField>

### Query Parameters

<ParamField query="page" type="number">
  Page number for pagination (default: 1)
</ParamField>

<ParamField query="pageSize" type="number">
  Number of results per page
</ParamField>

<ParamField query="search" type="string">
  Search term to filter links by name or URL
</ParamField>

### Response

<ResponseField name="data" type="array">
  Array of link objects
  
  <Expandable title="link object">
    <ResponseField name="id" type="string">
      Link UUID
    </ResponseField>
    
    <ResponseField name="name" type="string">
      Link name (max 100 characters)
    </ResponseField>
    
    <ResponseField name="url" type="string">
      Destination URL (max 500 characters)
    </ResponseField>
    
    <ResponseField name="slug" type="string">
      Unique URL slug (max 100 characters)
    </ResponseField>
    
    <ResponseField name="userId" type="string">
      Owner user UUID (for personal links)
    </ResponseField>
    
    <ResponseField name="teamId" type="string">
      Team UUID (for team links)
    </ResponseField>
    
    <ResponseField name="createdAt" type="string">
      ISO 8601 timestamp of creation
    </ResponseField>
    
    <ResponseField name="updatedAt" type="string">
      ISO 8601 timestamp of last update
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="count" type="number">
  Total number of links
</ResponseField>

<ResponseField name="page" type="number">
  Current page number
</ResponseField>

<ResponseField name="pageSize" type="number">
  Results per page
</ResponseField>

### Example Request

<CodeGroup>

```bash curl
curl "https://your-umami-instance.com/api/links?page=1&pageSize=20" \
  -H "Authorization: Bearer YOUR_TOKEN"
```

```javascript JavaScript
const response = await fetch(
  'https://your-umami-instance.com/api/links?page=1&pageSize=20',
  {
    headers: {
      'Authorization': 'Bearer YOUR_TOKEN',
    },
  }
);

const links = await response.json();
```

```python Python
import requests

response = requests.get(
    'https://your-umami-instance.com/api/links',
    params={'page': 1, 'pageSize': 20},
    headers={'Authorization': 'Bearer YOUR_TOKEN'}
)

links = response.json()
```

</CodeGroup>
